    // Enhanced validation function
    function validateInput(input, min, max, fieldName) {
        const value = parseFloat(input.value);
        const messageEl = document.getElementById(input.name.toLowerCase().replace('_', '') + '-message');
        
        if (!input.value) {
            if (messageEl) {
                messageEl.textContent = '';
                messageEl.className = 'validation-message';
            }
            return false;
        }
        
        if (isNaN(value) || value < min || value > max) {
            if (messageEl) {
                messageEl.textContent = `${fieldName} should be between ${min} and ${max}`;
                messageEl.className = 'validation-message error';
            }
            return false;
        } else {
            if (messageEl) {
                messageEl.textContent = `✓ Valid ${fieldName.toLowerCase()}`;
                messageEl.className = 'validation-message success';
            }
            return true;
        }
    }

    function displayComparisonResults(cars) {
        const resultsGrid = document.getElementById('resultsGrid');
        const statsGrid = document.getElementById('statsGrid');
        const recommendation = document.getElementById('recommendation');
        
        // Clear previous results
        resultsGrid.innerHTML = '';
        statsGrid.innerHTML = '';
        
        // Display each car's result
        cars.forEach((car, index) => {
            const carResult = document.createElement('div');
            carResult.className = 'car-result';
            carResult.innerHTML = `
                <div class="car-name">${car.car_name}</div>
                <div class="car-price">₹${(car.predicted_price/100000).toFixed(2)} lakhs</div>
                <div style="text-align: left; margin-top: 1rem;">
                    <div><strong>Brand:</strong> ${car.Brand}</div>
                    <div><strong>Age:</strong> ${car.Car_Age} years</div>
                    <div><strong>KM Driven:</strong> ${car.Kilometers_Driven.toLocaleString('en-IN')}</div>
                    <div><strong>Fuel:</strong> ${car.Fuel_Type}</div>
                    <div><strong>Transmission:</strong> ${car.Transmission}</div>
                </div>
            `;
            resultsGrid.appendChild(carResult);
        });
        
        // Calculate statistics
        const prices = cars.map(car => car.predicted_price);
        const highestPrice = Math.max(...prices);
        const lowestPrice = Math.min(...prices);
        const priceDifference = highestPrice - lowestPrice;
        const averagePrice = prices.reduce((sum, price) => sum + price, 0) / prices.length;
        
        // Display statistics
        const stats = [
            { label: 'Highest Price', value: `₹${(highestPrice/100000).toFixed(2)} lakhs` },
            { label: 'Lowest Price', value: `₹${(lowestPrice/100000).toFixed(2)} lakhs` },
            { label: 'Price Difference', value: `₹${(priceDifference/100000).toFixed(2)} lakhs` },
            { label: 'Average Price', value: `₹${(averagePrice/100000).toFixed(2)} lakhs` }
        ];
        
        stats.forEach(stat => {
            const statItem = document.createElement('div');
            statItem.className = 'stat-item';
            statItem.innerHTML = `
                <div class="stat-value">${stat.value}</div>
                <div class="stat-label">${stat.label}</div>
            `;
            statsGrid.appendChild(statItem);
        });
        
        // Generate recommendation
        const bestValueCar = cars.reduce((best, current) => 
            (current.predicted_price / current.Kilometers_Driven) < (best.predicted_price / best.Kilometers_Driven) ? current : best
        );
        
        recommendation.innerHTML = `
            <p>Based on price per kilometer, <strong>${bestValueCar.car_name}</strong> offers the best value at 
            ₹${(bestValueCar.predicted_price/100000).toFixed(2)} lakhs with ${bestValueCar.Kilometers_Driven.toLocaleString('en-IN')} km.</p>
        `;
        
        // Show results section
        document.getElementById('comparisonResults').style.display = 'block';
    }
